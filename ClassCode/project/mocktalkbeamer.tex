\begin{frame}{Research Puzzle}
\protect\phantomsection\label{research-puzzle}
\begin{columns}[T]
\begin{column}{0.55\linewidth}
\textbf{Motivating Question}

\begin{itemize}
\tightlist
\item
  Climate Change is an existential global challenge
\item
  Countries and people need to reduce their emissions \emph{yesterday}
\item
  Economic inequality leads to elites having more political power
\item
  National climate policies have a large impact on emissions
\end{itemize}
\end{column}

\begin{frame}{Research Question}
\protect\phantomsection\label{research-question}
\begin{columns}[T]
\begin{column}{0.55\linewidth}
\textbf{Motivating Question}

\begin{itemize}
\tightlist
\item
  Emissions are unequal within countries

  \begin{itemize}
  \tightlist
  \item
    Oxfam: The world's richest 1\% are responsible for more carbon
    emissions than the poorest 66\%
  \end{itemize}
\item
  \textbf{Do more unequal countries do less to mitigate climate change?}
\end{itemize}
\end{column}

\begin{column}{0.45\linewidth}
\pandocbounded{\includegraphics[keepaspectratio]{images/oxfam.png}}

\begin{frame}{Background and Literature}
\protect\phantomsection\label{background-and-literature}
\textbf{What we know}

\begin{itemize}
\tightlist
\item
  There is wide varition in global emissions
\item
  Much is driven by wealth

  \begin{itemize}
  \tightlist
  \item
    Wealth -\textgreater{} Consumption -\textgreater{} Emissions
  \end{itemize}
\item
  Generally, people are supportive of climate policies, but pushback
  from rich elites can hinder climate action
\end{itemize}

\note{Speaker notes: These don't show up on the screen, but can be
useful when building}
\end{frame}

\begin{frame}{Hypotheses}
\protect\phantomsection\label{hypotheses}
I derive the following expectations:

\begin{itemize}[<+->]
\item
  \textbf{H1 (Main):} Ceterus paribus, as \emph{Income inequality}
  increases, \emph{Per Capita carbon emissions} will also increase
\item
  \textbf{Null Hypothesis} There is no relationship between income
  inequality and carbon emissions.
\end{itemize}
\end{frame}

\begin{frame}{Contributions \& Further Research}
\protect\phantomsection\label{contributions-further-research}
\textbf{There does not appear to be a strong relationship between wealth
inquality and CO2 Emissions}

\begin{itemize}
\tightlist
\item
  Potential Omitted Variable Bias

  \begin{itemize}
  \tightlist
  \item
    Political Insitutions - Democracies and Interest group systems - ie
    how elites preferences are heard
  \item
    Oil Dependent Economies
  \item
    Regional Differences
  \end{itemize}
\item
  Could look at climate policy as DV, rather than emissions
\end{itemize}
\end{frame}

\begin{frame}{Contributions \& Further Research}
\protect\phantomsection\label{contributions-further-research-1}
\textbf{Qualitative Follow-Up}

To probe the mechanism, future work should conduct process tracing to
better understand if or how economic elites shape climate policy.

\begin{itemize}
\tightlist
\item
  United States as a most crucial case
\item
  Thailand as ``typical'' case
\end{itemize}

Case studies of wealthy people/companies in each country
\end{frame}

\begin{frame}{Discussion}
\protect\phantomsection\label{discussion}
\textbf{Policy Implications}

These results suggest wealth inequality is not a meaningful impediment
to effective climate policy

\textbf{Limitations}

\begin{itemize}
\tightlist
\item
  One year snapshot
\item
  Country Level of analysis
\item
  Aggrgate Numbers
\end{itemize}
\end{frame}
